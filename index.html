export default async function handler(req, res) { const targetUrl = req.query.url; if (!targetUrl) { return res.status(400).send("URL 필요"); } try { const response = await fetch(targetUrl, { headers: { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 Chrome/119 Safari/537.36" } }); const contentType = response.headers.get("content-type"); res.setHeader("Content-Type", contentType); const data = await response.text(); res.send(data); } catch (err) { res.status(500).send("에러 발생: " + err.message); } }
